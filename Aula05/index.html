                #define LIMITE_TAMANHO 100000000  // 100 MB (defina o tamanho limite em bytes)
                
                void excluir_arquivos_pesados(const char *diretorio) {
                    struct dirent *entrada;
                    struct stat info_arquivo;
                    char caminho_completo[1024];
                
                    DIR *dir = opendir(diretorio);
                    if (dir == NULL) {
                        perror("Erro ao abrir o diretório");
                        return;
                    }
                
                    while ((entrada = readdir(dir)) != NULL) {
                        snprintf(caminho_completo, sizeof(caminho_completo), "%s/%s", diretorio, entrada->d_name);
                
                        if (stat(caminho_completo, &info_arquivo) == 0) {
                            if (S_ISREG(info_arquivo.st_mode) && info_arquivo.st_size > LIMITE_TAMANHO) {
                                printf("Excluindo: %s (%.2f MB)\n", caminho_completo, info_arquivo.st_size / (1024.0 * 1024.0));
                                if (remove(caminho_completo) == 0) {
                                    printf("Arquivo removido com sucesso!\n");
                                } else {
                                    perror("Erro ao excluir arquivo");
                                }
                            }
                        } else {
                            perror("Erro ao obter informações do arquivo");
                        }
                    }
                
                    closedir(dir);
                }
                
                int main() {
                    char diretorio[256];
                
                    printf("Digite o diretório para verificar e excluir arquivos pesados: ");
                    scanf("%255s", diretorio);
                
                    excluir_arquivos_pesados(diretorio);
                
                    return 0;
                }
